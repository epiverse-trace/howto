How to estimate the daily growth rate using the date of report stratified by groups?

Published

July 18, 2023

library(tidyverse)
library(covidregionaldata)
library(i2extras)
library(incidence2)

raw_dat <- covidregionaldataUK %>% as_tibble() %>% 
  filter(magrittr::is_in(region,c("England", "Scotland", 
                                  "Northern Ireland", "Wales"))) %>% 
  filter(date > lubridate::ymd(20200701)) %>%
  filter(date < lubridate::ymd(20201101))

raw_dat %>%
  ggplot() +
  aes(x = date, y = cases_new, col = region) +
  geom_line(alpha = 0.4)

# Convert the data to an incidence object
dat <- incidence(
  x = raw_dat, 
  counts = "cases_new",
  date_index = "date",
  interval = "day"
)

# Stratify analysis by groups

  x = raw_dat, 
  counts = "cases_new",
  date_index = "date",
  interval = "day",
  groups = "region"
